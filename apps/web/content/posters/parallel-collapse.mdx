frag: |
  precision highp float;

  uniform float uTime;
  uniform vec2 uResolution;

  void main() {
    vec2 uv = (gl_FragCoord.xy - 0.5 * uResolution) / min(uResolution.x, uResolution.y);

    float t = uTime * 0.5;
    vec2 p1 = vec2(cos(t), sin(t)) * 0.5;
    vec2 p2 = vec2(cos(t + 3.14159), sin(t + 3.14159)) * 0.5;

    float d1 = length(uv - p1);
    float d2 = length(uv - p2);

    float wave = sin(d1 * 20.0 - uTime * 3.0) * sin(d2 * 20.0 - uTime * 3.0);
    vec3 color = vec3(0.5) + vec3(0.4, 0.2, 0.6) * wave;

    gl_FragColor = vec4(color, 1.0);
  }

Multiple timelines diverge. Every decision spawns alternatives. Quantum superposition until observation forces collapse. We exist in parallel until context demands we choose one version of ourselves.
