  precision highp float;

  uniform float uTime;
  uniform vec2 uResolution;

  }

  void main() {
    vec2 uv = gl_FragCoord.xy / uResolution;

    float scanline = sin(uv.y * 800.0 + uTime * 5.0) * 0.05;
    float noise = random(uv + fract(uTime)) * 0.3;
    float wave = sin(uv.x * 20.0 + uTime * 3.0) * 0.1;

    float signal = scanline + noise + wave;
    vec3 color = vec3(signal) + vec3(0.2, 0.3, 0.3);

    gl_FragColor = vec4(color, 1.0);
  }
