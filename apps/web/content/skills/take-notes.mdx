  precision highp float;
  uniform float uTime;
  uniform vec2 uResolution;

  float hash(vec2 p) {
    return fract(sin(dot(p, vec2(127.1, 311.7))) * 43758.5453);
  }

  void main() {
    float size = 40.0;
    float height = size * 0.866;  // sqrt(3)/2 for equilateral

    vec2 p = gl_FragCoord.xy;
    float row = floor(p.y / height);
    float col = floor(p.x / size);

    // Which half of the cell: left (up) or right (down)
    float localX = mod(p.x, size);
    float localY = mod(p.y, height);

    // Up triangle if in left half and below/above midline appropriately
    // Down triangle if in right half
    float isUp = step(localX, size * 0.5);
    float isDown = 1.0 - isUp;

    // Flip row for down triangles to interlock
    float triRow = mix(row * 2.0, row * 2.0 + 1.0, isDown);

    // Triangle ID for hash
    vec2 triId = vec2(col, triRow);

    // Checker: alternate based on row
    float checker = mod(triRow + col, 2.0);

    // Animation
    float cellHash = hash(floor(triId));
    float flipInterval = 15.0 + cellHash * 15.0;
    float phase = mod(uTime + cellHash * 100.0, flipInterval) / flipInterval;
    float isFlipping = step(0.49, phase) * step(phase, 0.51);

    checker = mix(checker, 1.0 - checker, isFlipping);

    gl_FragColor = vec4(vec3(checker), 1.0);
  }
