fragmentShader: |
  precision highp float;
  uniform float uTime;
  uniform vec2 uResolution;

  float hash(vec2 p) {
    return fract(sin(dot(p, vec2(127.1, 311.7))) * 43758.5453);
  }

  void main() {
    float cellSize = 50.0;
    vec2 cell = floor(gl_FragCoord.xy / cellSize);
    float checker = mod(cell.x + cell.y, 2.0);

    // Each cell flips independently on a random schedule
    // Using cell hash as seed for its unique timing
    float cellHash = hash(cell);

    // Time between flips for this cell: 5-15 seconds
    float flipInterval = 15.0 + cellHash * 15.0;

    // Phase within current interval
    float phase = mod(uTime + cellHash * 100.0, flipInterval) / flipInterval;

    // Flip when phase crosses 0.5 (random moment in each cell's cycle)
    // Small window so only 1-2 cells flip at a time
    float isFlipping = step(0.49, phase) * step(phase, 0.51);

    checker = mix(checker, 1.0 - checker, isFlipping);

    gl_FragColor = vec4(vec3(checker), 1.0);
  }
